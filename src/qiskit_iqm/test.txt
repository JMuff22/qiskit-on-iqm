from qiskit import QuantumCircuit
from qiskit_iqm import IQMFakeAdonis
from qiskit.compiler import transpile

circuit = QuantumCircuit(2, 2)
circuit.x(1)
circuit.measure(0, 0)


backend = IQMFakeAdonis()
circuit = transpile(circuit, basis_gates=backend.basis_gates)
job = backend.run([circuit], qubit_mapping=None, shots=10)
job.result().get_counts()

from qiskit import execute
from qiskit import QuantumCircuit
from qiskit_iqm import IQMFakeAdonis
from qiskit.compiler import transpile

circuit = QuantumCircuit(2, 2)
circuit.x(1)
circuit.measure(0, 0)


backend = IQMFakeAdonis()
circuit = transpile(circuit, basis_gates=backend.basis_gates)
job = execute(circuit, backend, qubit_mapping=None, shots=10)
job.result().get_counts()
